\begin{proposition}[Local curvature invariant via global zero distribution]
\label{prop:hadamard_A_decomp}
Let $\rho_n = \tfrac{1}{2} + i t_n$ ($t_1 < t_2 < \cdots$, $t_n > 0$) denote the
non-trivial zeros of $\zeta(s)$ on the critical line (assuming RH).
Fix $\rho_0 = \tfrac{1}{2} + i t_0$ and write $\bar\rho_0 = \tfrac{1}{2} - i t_0$.
Then the local curvature coefficient
\[
  A(t_0) \;=\; \lim_{\delta\to 0}\!\left[\kappa(\rho_0{+}\delta) - \tfrac{1}{\delta^2}\right],
  \quad \kappa(s) = |\xi'/\xi(s)|^2,
\]
admits the \emph{Hadamard decomposition}
\[
  A(t_0) \;=\; B(t_0)^2 + 2H_1(t_0),
\]
where
\begin{align*}
  B(t_0)  &= -\frac{1}{2t_0}
              - \sum_{n:\,t_n \neq t_0}
                \!\left[\frac{1}{t_0-t_n} + \frac{1}{t_0+t_n}\right], \\
  H_1(t_0)&= \phantom{-}\frac{1}{4t_0^2}
              + \sum_{n:\,t_n \neq t_0}
                \!\left[\frac{1}{(t_0-t_n)^2} + \frac{1}{(t_0+t_n)^2}\right].
\end{align*}
Both series converge absolutely (terms $\sim 2t_0/t_n^2$ and $\sim 2/t_n^2$
respectively as $n \to \infty$).
\end{proposition}

\begin{proof}
\textbf{Step 1 (Hadamard product, $D=0$).}
Since $\Sigma_n \operatorname{Re}(2/\rho_n) = \Sigma_n 1/({\tfrac14+t_n^2}) < \infty$,
the product $\Pi_n\!\left[(1-s/\rho_n)(1-s/\bar\rho_n)\right]$ converges absolutely.
Write $\xi(s) = \xi(0)\,e^{Ds}\,\Pi_n[(1-s/\rho_n)(1-s/\bar\rho_n)]$.
The symmetry $\xi(s)=\xi(1-s)$ gives $\xi'/\xi(\tfrac12)=0$.
Evaluating the log-derivative at $s=\tfrac12$ yields
$D + \Sigma_n\!\left[\tfrac{1}{\frac12-\rho_n}+\tfrac{1}{\frac12-\bar\rho_n}\right] = 0$;
each paired term equals $\tfrac{1}{-it_n}+\tfrac{1}{it_n}=0$, so $D=0$.
Hence
\[
  \xi'/\xi(s) = \sum_{n\ge 1}\!\left[\frac{1}{s-\rho_n}+\frac{1}{s-\bar\rho_n}\right].
\]

\textbf{Step 2 (Laurent expansion at $\rho_0$).}
Isolating the $\rho_0$ pole and writing $g(s) = \xi'/\xi(s) - 1/(s-\rho_0)$:
\[
  g(s) = \frac{1}{s-\bar\rho_0}
         + \sum_{n:\,t_n\neq t_0}\!\!\left[\frac{1}{s-\rho_n}+\frac{1}{s-\bar\rho_n}\right].
\]
Each term $\tfrac{1}{s-\rho_n}+\tfrac{1}{s-\bar\rho_n}$ at $s=\rho_0$ equals
$\tfrac{1}{i(t_0-t_n)}+\tfrac{1}{i(t_0+t_n)}$, which is purely imaginary.
Hence $\operatorname{Re}(g(\rho_0))=0$, confirming $c_0 = g(\rho_0) = i B(t_0)$.
Differentiating $g$ and evaluating at $\rho_0$:
$c_1 = g'(\rho_0) = \tfrac{1}{4t_0^2} + \Sigma_{n\neq 0}\!\left[\tfrac{1}{(t_0-t_n)^2}+\tfrac{1}{(t_0+t_n)^2}\right] = H_1(t_0)\in\mathbb{R}$.

\textbf{Step 3 (Curvature expansion).}
With $c_0=iB$, $c_1=H_1\in\mathbb{R}$:
\[
  \kappa(\rho_0+\delta)
  = \left|\tfrac1\delta + iB + H_1\delta + O(\delta^2)\right|^2
  = \tfrac{1}{\delta^2} + B^2 + 2H_1 + O(\delta^2).
\]
Taking $\delta\to 0$ gives $A(t_0) = B(t_0)^2 + 2H_1(t_0)$.
\end{proof}

\begin{remark}[Numerical verification]
For the 13 non-trivial zeros $t_n \in [14.13, 59.35]$, we compute both sides:
\begin{itemize}
\item \emph{Direct}: $A_{\mathrm{dir}}(t_0) = \kappa(\rho_0+\delta)-1/\delta^2$ at $\delta=0.01$.
\item \emph{Hadamard}: $A_{\mathrm{Had}}(N) = B_N^2 + 2H_{1,N}$, partial sums with $N=2000$ zeros
  plus Euler–Maclaurin tail correction
  $B_{\mathrm{tail}} = \int_{t_N}^\infty \tfrac{2t_0}{t^2-t_0^2}\tfrac{\log(t/2\pi)}{2\pi}\,dt$.
\end{itemize}
With tail correction, $|A_{\mathrm{Had}} - A_{\mathrm{dir}}|/A_{\mathrm{dir}} < 0.003\%$
for all 13 zeros ($13/13$).
\end{remark}
